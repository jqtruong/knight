;; Change according to color
(defparameter *up* 1)
(defvar *moves-ruleset* 
  (knight
   ;; n
   (1 (up 2)) (-1 (up 2))
   ;; e
   (2 (up)) (2 (down))
   ;; s
   (1 (down 2)) (-1 (down 2))
   ;; w
   (-2 (up)) (-2 (down)))

  (pawn
   (normal (0 (up)))
   (special-startup (0 (up)) (0 ((up) 2)))
   (;; should be a condition for en-passant
    attack (1 (up))
           (-1 (up))))

  (king
   ;; clockwise
   (0 (up))
   (1 (up))
   (1 0)
   (1 (down))
   (0 (down))
   (-1 (down))
   (-1 0)
   (-1 (up)))

  (queen
   ;; diagonals
   (x (up x)) (x (down x)) (-x (up x)) (-x (down x))
   ;; up down
   (0 (up x)) (0 (down x))
   ;; left right
   (-x 0) (x 0))

  (bishop
   ;; diagonals
   (x (up x)) (x (down x)) (-x (up x)) (-x (down x)))

  (rook 
   ;; up down
   (0 (up x)) (0 (down x))
   ;; left right
   (-x 0) (x 0)))
